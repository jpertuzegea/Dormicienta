CREATE DATABASE SONOLIENTA;
USE SONOLIENTA;

CREATE TABLE USUARIO(){
USUARIOID 
NOMBRE,
USUARIORED,
CLAVE,
ESTADO
}

CREATE TABLE CATEGORIA(){
CATEGORIAID INT IDENTITY(1,1),
NOMBRE,
DESCRIPCION, 
IMAGEN,
ESTADO,
PRIMARY KEY (CATEGORIAID)
}

CREATE TABLE MARCA(){
MARCAID 
NOMBRE,
ESTADO
}

CREATE TABLE PRODUCTO(){
PRODUCTOID INT IDENTITY(1,1),
NOMBRE,
CODIGO,
DESCRIPCION, 
PRECIO,
CATEGORIA,
CANTIDAD_DISPONIBLE,
IMAGEN,
ESTADO,
PRIMARY KEY (PRODUCTOID)
}
